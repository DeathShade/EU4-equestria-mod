		NOT = { has_country_flag = had_selenite_reformation }

		religion = celestian
		num_of_cities = 10
	}

	
	
	can_start = {
		has_any_disaster = no
		OR =
		{
			NOT = { stability = 0 }
			war_exhaustion = 3
		}
	}
	progress = {
		modifier = {
			factor = 0.3
			war_exhaustion = 3
		}
		modifier = {
			factor = 0.5
			war_exhaustion = 5
		}
		modifier = {
			factor = 0.5
			war_exhaustion = 10
		}
		modifier = {
			factor = 1
			war_exhaustion = 15
		}
		modifier = {
			factor = 1.5
			war_exhaustion = 20
		}
		modifier = {
			factor = 1
			is_bankrupt = yes
		} 
		modifier = {
			factor = 1
			has_idea_group = religious_ideas
		}

	}
	
	can_end = { #note : we're not supposed to be using that, there are events (pony.7,10,11) for that kind of things !
		NOT = { num_of_revolts = 1 }
		NOT = {
			has_spawned_rebels = heretic_rebels
		}
		stability = 3
		NOT =  { war_exhaustion = 1 }
		NOT = {
				any_owned_province = {
					has_owner_religion = no
				}		
		}
	}
	
	modifier = {
		global_unrest = 10
		legitimacy = -2
	}

	}
